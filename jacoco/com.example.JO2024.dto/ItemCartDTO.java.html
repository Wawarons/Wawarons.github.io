package com.example.JO2024.dto;

import java.math.BigDecimal;

import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Positive;
import jakarta.validation.constraints.Size;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Represents an item in a shopping cart with the following properties:
 * - id: a unique identifier for the item
 * - title: the title of the item (3 to 25 characters long)
 * - description: a description of the item
 * - price: the price of the item (must be positive)
 * - quantity: the quantity of the item in the cart (must be positive)
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Setter
public class ItemCartDTO {

    @NotNull
    private long id;

    @NotNull
    @NotEmpty
    @Size(min = 3, max = 25)
    private String title;

    @NotNull
    private String description;

    @NotNull
    @Positive
    private BigDecimal price;

    @NotNull
    @Positive
    private int quantity;

}
