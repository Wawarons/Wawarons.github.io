package com.example.JO2024.dto;

import java.math.BigDecimal;
import java.time.Instant;

import com.fasterxml.jackson.annotation.JsonProperty;

import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Positive;
import jakarta.validation.constraints.Size;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Represents a Data Transfer Object (DTO) for a ticket.
 * This DTO includes information such as the ticket ID, title, description,
 * quantity, price,
 * availability status, start date, and end date.
 * The class provides constructors to create a TicketDTO object with different
 * combinations of parameters.
 * It also includes getter and setter methods for each field, as well as a
 * builder for easy object creation.
 */
@NoArgsConstructor
@AllArgsConstructor
@Getter
@Setter
@Builder
public class TicketDTO {

    private long id;

    @NotEmpty
    @NotNull
    @Size(min = 3, max = 25)
    private String title;

    @Size(min = 5, max = 250)
    private String description;

    @NotNull
    @Positive
    private int quantity;

    @NotNull
    @Positive
    private BigDecimal price;

    @NotNull
    @JsonProperty("isAvailable")
    private boolean isAvailable;

    @NotNull
    private Instant startDate;

    @NotNull
    private Instant endDate;

    public TicketDTO(@NotEmpty @NotNull @Size(min = 3, max = 25) String title,
            @Size(min = 5, max = 250) String description, @NotNull @Positive int quantity,
            @NotNull @Positive BigDecimal price, @NotNull boolean isAvailable, @NotNull Instant startDate,
            @NotNull Instant endDate) {
        this.title = title;
        this.description = description;
        this.quantity = quantity;
        this.price = price;
        this.isAvailable = isAvailable;
        this.startDate = startDate;
        this.endDate = endDate;
    }

    public TicketDTO(@NotEmpty @NotNull @Size(min = 3, max = 25) String title,
            @Size(min = 5, max = 250) String description,
            @NotNull @Positive BigDecimal price, @NotNull Instant startDate,
            @NotNull Instant endDate) {
        this.title = title;
        this.description = description;
        this.price = price;
        this.startDate = startDate;
        this.endDate = endDate;
    }

}
