package com.example.JO2024.dto;

import com.example.JO2024.errors.ErrorMessages;

import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Pattern;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Represents a Data Transfer Object (DTO) for password information.
 * This class includes validation annotations for ensuring the password meets
 * specific criteria:
 * - Not empty
 * - Not null
 * - Matches a specific pattern:
 * - At least one lowercase letter
 * - At least one uppercase letter
 * - At least one digit
 * - At least one special character from the set [.*$_!\\-+@;:/\\\\|']
 * - Length between 8 and 20 characters
 * If the password does not meet the specified format, an error message is
 * provided.
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Setter
public class PasswordDTO {

    @NotEmpty
    @NotNull
    @Pattern(regexp = "(?=.*[a-z]+)(?=.*[A-Z]+)(?=.*[0-9]+)(?=.*[.*$_!\\-+@;:/\\\\|']+)[A-Za-z0-9.*$_!\\-+@;:/\\\\|']+", message = ErrorMessages.PASSWORD_WRONG_FORMAT)
    private String password;

}
