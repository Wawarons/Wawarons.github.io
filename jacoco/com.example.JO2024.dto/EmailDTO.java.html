package com.example.JO2024.dto;

import jakarta.validation.constraints.Email;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.NotNull;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * A data transfer object (DTO) class representing an email address.
 * It includes validation annotations for ensuring the email field is not empty,
 * not null, not blank, and a valid email format.
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Setter
public class EmailDTO {

    @NotEmpty
    @NotNull
    @NotBlank
    @Email
    private String email;
}
