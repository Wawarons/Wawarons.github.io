package com.example.JO2024.dto;

import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Positive;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Represents a Data Transfer Object (DTO) for a user ID.
 * This class includes a single field 'userId' that is annotated with @NotNull
 * and @Positive to ensure it is not null and has a positive value.
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Setter
public class UserIdDTO {

    @NotNull
    @Positive
    private long userId;

}
