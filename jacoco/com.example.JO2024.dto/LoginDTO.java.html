package com.example.JO2024.dto;

import com.example.JO2024.errors.ErrorMessages;

import jakarta.validation.constraints.Email;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Pattern;
import lombok.AllArgsConstructor;

/**
 * The LoginDTO class represents a data transfer object for login information.
 * It contains the email address and password provided by the customer for
 * authentication.
 */
@AllArgsConstructor
public class LoginDTO {

  @NotNull
  @Email
  private String email;

  @NotNull
  @Pattern(regexp = "(?=.*[a-z]+)(?=.*[A-Z]+)(?=.*[0-9]+)(?=.*[.*$_!\\-+@;:/\\\\|']+)[A-Za-z0-9.*$_!\\-+@;:/\\\\|']+", message = ErrorMessages.PASSWORD_WRONG_FORMAT)
  private String password;

  /**
   * Get email address send by the customer for the authentification.
   * 
   * @return Email address send by the customer for the authentification.
   */
  public String getEmail() {
    return email;
  }

  /**
   * Get the password send by the customer for the authentification.
   * 
   * @return Password send by the customer for the authentification.
   */
  public String getPassword() {
    return password;
  }

}
