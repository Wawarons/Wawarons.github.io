package com.example.JO2024.dto;

import com.example.JO2024.errors.ErrorMessages;

import jakarta.validation.constraints.Email;
import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Pattern;
import jakarta.validation.constraints.Size;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * The SignupCustomerDTO class represents a data transfer object for signing up
 * a customer.
 * It contains the following fields:
 * - email: a string representing the customer's email address
 * - password: a string representing the customer's password
 * - name: a string representing the customer's name
 * - firstname: a string representing the customer's first name
 * 
 * This class is used for validating and transferring customer signup data.
 */
@Builder
@Getter
@Setter
@AllArgsConstructor
@NoArgsConstructor
public class SignupCustomerDTO {

  @NotEmpty
  @NotNull
  @Email
  private String email;

  @NotEmpty
  @NotNull
  @Pattern(regexp = "(?=.*[a-z]+)(?=.*[A-Z]+)(?=.*[0-9]+)(?=.*[.*$_!\\-+@;:/\\\\|']+)[A-Za-z0-9.*$_!\\-+@;:/\\\\|']+", message = ErrorMessages.PASSWORD_WRONG_FORMAT)
  private String password;

  @NotEmpty
  @NotNull
  @Size(min = 2, max = 50, message = ErrorMessages.NAME_FORMAT)
  private String name;

  @NotEmpty
  @NotNull
  @Size(min = 2, max = 50, message = ErrorMessages.FIRSTNAME_FORMAT)
  private String firstname;

}
