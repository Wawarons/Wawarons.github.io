package com.example.JO2024.dto;

import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Positive;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Represents a Data Transfer Object (DTO) for a customer.
 * Contains fields for customer information such as id, name, firstname, email,
 * block status, verification status, and customer key.
 * Includes annotations for validation constraints using Jakarta Bean Validation
 * API.
 * Utilizes Lombok annotations for generating constructors, builder, getters,
 * and setters.
 */
@NoArgsConstructor
@AllArgsConstructor
@Builder
@Getter
@Setter
public class CustomerDTO {

    @NotNull
    @Positive
    private long id;

    @NotNull
    @NotEmpty
    private String name;

    @NotNull
    @NotEmpty
    private String firstname;

    @NotNull
    @NotEmpty
    private String email;

    @NotNull
    private boolean isBlock;

    @NotNull
    private boolean isVerified;

    @NotNull
    @NotEmpty
    private String customerKey;

}
