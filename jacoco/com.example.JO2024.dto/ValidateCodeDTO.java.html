package com.example.JO2024.dto;

import com.example.JO2024.errors.ErrorMessages;

import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Pattern;
import jakarta.validation.constraints.Positive;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * The ValidateCodeDTO class represents a data transfer object for validating a
 * code.
 * It contains the following properties:
 * - userId: a long value representing the user ID (must be not null and
 * positive)
 * - code: a string representing the code (must be not null and have a length of
 * 6 digits)
 * 
 * This class is used for validating codes in the JO2024 application.
 */
@AllArgsConstructor
@NoArgsConstructor
@Builder
@Getter
@Setter
public class ValidateCodeDTO {

    @NotNull
    @Positive
    private long userId;

    @NotNull
    @Pattern(regexp = "[0-9]{6}", message = ErrorMessages.CODE_WRONG_SIZE)
    private String code;

}
