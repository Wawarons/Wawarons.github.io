package com.example.JO2024.services;

import java.time.Instant;
import java.util.List;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.JO2024.dto.TicketDTO;
import com.example.JO2024.models.Ticket;
import com.example.JO2024.repositories.TicketRepository;

/**
 * Service class for managing Ticket entities.
 * Provides methods to create, delete, retrieve, and update Ticket records.
 */
@Service
public class TicketService {

    @Autowired
    private TicketRepository ticketRepository;

    /**
     * Retrieves a Ticket by its ID.
     * 
     * @param id the ID of the Ticket
     * @return an Optional containing the Ticket if found, or an empty Optional if
     *         not found
     */
    public Optional<Ticket> getTicketById(long id) {
        return ticketRepository.findById(id);
    }

    /**
     * Saves a new Ticket entity to the repository.
     * 
     * @param ticket the Ticket entity to be saved
     */
    public void create(Ticket ticket) {
        ticketRepository.save(ticket);
    }

    /**
     * Deletes a Ticket entity from the repository.
     * 
     * @param ticket the Ticket entity to be deleted
     */
    public void delete(Ticket ticket) {
        ticketRepository.delete(ticket);
    }

    /**
     * Retrieves all Ticket entities from the repository.
     * 
     * @return a list of all Ticket entities
     */
    public List<Ticket> getAllTickets() {
        return ticketRepository.findAll();
    }

    /**
     * Retrieves all Ticket entities ordered by price in descending order.
     * 
     * @return a list of Ticket entities ordered by price descending
     */
    public List<Ticket> getAllTicketsByPriceDesc() {
        return ticketRepository.findAllByOrderByPriceDesc();
    }

    /**
     * Retrieves all Ticket entities ordered by price in ascending order.
     * 
     * @return a list of Ticket entities ordered by price ascending
     */
    public List<Ticket> getAllTicketsByPriceAsc() {
        return ticketRepository.findAllByOrderByPriceAsc();
    }

    /**
     * Retrieves all Ticket entities ordered by start date in descending order.
     * 
     * @return a list of Ticket entities ordered by start date descending
     */
    public List<Ticket> getAllTicketsByStartDateDesc() {
        return ticketRepository.findAllByOrderByStartDateDesc();
    }

    /**
     * Retrieves all Ticket entities ordered by start date in ascending order.
     * 
     * @return a list of Ticket entities ordered by start date ascending
     */
    public List<Ticket> getAllTicketsByStartDateAsc() {
        return ticketRepository.findAllByOrderByStartDateAsc();
    }

    /**
     * Retrieves all Ticket entities ordered by quantity in descending order.
     * 
     * @return a list of Ticket entities ordered by quantity descending
     */
    public List<Ticket> getAllTicketsByQuantityDesc() {
        return ticketRepository.findAllByOrderByQuantityDesc();
    }

    /**
     * Retrieves all Ticket entities ordered by quantity in ascending order.
     * 
     * @return a list of Ticket entities ordered by quantity ascending
     */
    public List<Ticket> getAllTicketsByQuantityAsc() {
        return ticketRepository.findAllByOrderByQuantityAsc();
    }

    /**
     * Updates an existing Ticket entity in the repository.
     * 
     * @param ticket the Ticket entity to be updated
     */
    public void update(Ticket ticket) {
        ticketRepository.save(ticket);
    }

    /**
     * Updates a Ticket entity with values from a TicketDTO.
     * 
     * @param ticket    the Ticket entity to be updated
     * @param ticketDTO the TicketDTO containing new values
     * @return the updated Ticket entity
     */
    public Ticket updateByTicketDTO(Ticket ticket, TicketDTO ticketDTO) {
        ticket.setAvailable(ticketDTO.isAvailable());
        ticket.setDescription(ticketDTO.getDescription());
        ticket.setTitle(ticketDTO.getTitle());
        ticket.setStartDate(ticketDTO.getStartDate());
        ticket.setEndDate(ticketDTO.getEndDate());
        ticket.setPrice(ticketDTO.getPrice());
        ticket.setQuantity(ticketDTO.getQuantity());

        return ticket;
    }

    /**
     * Checks if a Ticket is available for purchase.
     * 
     * @param ticket the Ticket entity to be checked
     * @return true if the Ticket is available for purchase, false otherwise
     */
    public boolean isTicketAvailableToPurchase(Ticket ticket) {
        return ticket.isAvailable() && ticket.getQuantity() > 0
                && ticket.getEndDate().isAfter(Instant.now());
    }
}
