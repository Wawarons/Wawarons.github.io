package com.example.JO2024.services;

import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.JO2024.models.RefreshToken;
import com.example.JO2024.repositories.RefreshTokenRepository;

@Service
public class RefreshTokenService {

    @Autowired
    private RefreshTokenRepository refreshTokenRepository;

    /**
     * Retrieves the refresh token with the specified ID.
     *
     * @param id the ID of the refresh token to retrieve
     * @return an Optional containing the refresh token, or an empty Optional if no
     *         refresh token is found
     */
    public Optional<RefreshToken> getRefreshTokenById(long id) {
        return refreshTokenRepository.findById(id);
    }

    /**
     * Retrieves the refresh token with the specified customer ID.
     *
     * @param id the customer ID of the refresh token to retrieve
     * @return an Optional containing the refresh token, or an empty Optional if no
     *         refresh token is found
     */
    public Optional<RefreshToken> getRefreshTokenByCustomerId(long id) {
        return refreshTokenRepository.findRefreshTokenByCustomerId(id);
    }

    /**
     * Adds a refresh token to the repository.
     *
     * @param refreshToken the refresh token to add
     */
    public void addRefreshToken(RefreshToken refreshToken) {
        refreshTokenRepository.save(refreshToken);
    }

    /**
     * Checks if a refresh token exists for the specified customer ID.
     *
     * @param id the customer ID to check for a refresh token
     * @return true if a refresh token exists for the customer ID, false otherwise
     */
    public boolean hasRefreshTokenByCustomerId(long id) {
        return refreshTokenRepository.findRefreshTokenByCustomerId(id).isPresent();
    }

}
