package com.example.JO2024.services;

import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.JO2024.models.ResetPasswordToken;
import com.example.JO2024.repositories.ResetPasswordTokenRepository;

@Service
public class ResetPasswordTokenService {

    @Autowired
    private ResetPasswordTokenRepository resetPasswordTokenRepository;

    /**
     * Saves the provided ResetPasswordToken by calling the save method of the
     * ResetPasswordTokenRepository.
     *
     * @param resetPasswordToken the ResetPasswordToken to be saved
     */
    public void createResetTokenPassword(ResetPasswordToken resetPasswordToken) {
        resetPasswordTokenRepository.save(resetPasswordToken);

    }

    /**
     * Updates the provided ResetPasswordToken by calling the save method of the
     * ResetPasswordTokenRepository.
     *
     * @param resetPasswordToken the ResetPasswordToken to be updated
     */
    public void updateResetTokenPassword(ResetPasswordToken resetPasswordToken) {

        resetPasswordTokenRepository.save(resetPasswordToken);

    }

    /**
     * Retrieves the ResetPasswordToken associated with the provided customer ID by
     * calling the
     * findResetPasswordTokenByCustomerId method of the
     * ResetPasswordTokenRepository.
     *
     * @param id the ID of the customer for which to retrieve the ResetPasswordToken
     * @return an Optional containing the ResetPasswordToken if found, empty
     *         otherwise
     */
    public Optional<ResetPasswordToken> getResetTokenByCustomerId(long id) {
        return resetPasswordTokenRepository.findResetPasswordTokenByCustomerId(id);
    }

    /**
     * Retrieves the last ResetPasswordToken associated with the provided token by
     * calling the findResetPasswordTokenByToken method of the
     * ResetPasswordTokenRepository.
     *
     * @param token the token for which to retrieve the last ResetPasswordToken
     * @return an Optional containing the last ResetPasswordToken if found, empty
     *         otherwise
     */
    public Optional<ResetPasswordToken> getLastResetPasswordTokenByToken(String token) {
        return resetPasswordTokenRepository.findResetPasswordTokenByToken(token);
    }
}
