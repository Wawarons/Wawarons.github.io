package com.example.JO2024.services;

import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.JO2024.models.Cart;
import com.example.JO2024.models.Customer;
import com.example.JO2024.repositories.CartRepository;

import jakarta.transaction.Transactional;

@Service
public class CartService {

    @Autowired
    public CartRepository cartRepository;

    /**
     * Saves the specified cart to the database.
     *
     * @param cart the cart to be saved
     */
    public void createNewCart(Cart cart) {
        cartRepository.save(cart);
    }

    /**
     * Deletes the specified cart.
     *
     * @param cart the cart to be deleted
     */
    @Transactional
    public void deleteCart(Cart cart) {
    
        cartRepository.delete(cart);
    }

    /**
     * Retrieves a cart by the ID.
     *
     * @param id the ID of the cart to retrieve
     * @return an Optional containing the cart with the given ID, or empty if not
     *         found
     */
    public Optional<Cart> getCartById(long id) {
        return cartRepository.findById(id);
    }

    /**
     * Retrieves a cart by the customer ID.
     *
     * @param id the customer ID to search for
     * @return an Optional containing the cart associated with the given customer
     *         ID, or empty if not found
     */
    public Optional<Cart> getCartByCustomerId(long id) {
        return cartRepository.findByCustomerId(id);
    }

    /**
     * Creates a new cart for the specified customer.
     *
     * @param customer the customer for whom the cart is created
     */
    public void createCartForCustomer(Customer customer) {
        Cart cart = new Cart();
        cart.setCustomer(customer);
        cartRepository.save(cart);
    }

}
