package com.example.JO2024.services;

import com.example.JO2024.models.Bundle;
import com.example.JO2024.repositories.BundleRepository;

import jakarta.transaction.Transactional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Optional;

@Service
public class BundleService {

    @Autowired
    private BundleRepository bundleRepository;

    /**
     * Retrieve all bundles from the database.
     *
     * @return A list of all bundles stored in the database.
     */
    public List<Bundle> getAllBundles() {
        return bundleRepository.findAll();
    }

    /**
     * Retrieve a specific bundle from the database by its ID.
     *
     * @param id The ID of the bundle to retrieve.
     * @return An Optional containing the bundle if found, empty otherwise.
     */
    public Optional<Bundle> getBundleById(Long id) {
        return bundleRepository.findById(id);
    }

    /**
     * Create a new bundle in the database.
     *
     * @param bundle The bundle object to be created.
     * @return The created bundle object.
     */
    public Bundle createBundle(Bundle bundle) {
        return bundleRepository.save(bundle);
    }

    /**
     * Update a specific bundle in the database by its ID with the provided details.
     *
     * @param id            The ID of the bundle to update.
     * @param bundleDetails The updated details of the bundle.
     * @return The updated bundle object if the update was successful, null
     *         otherwise.
     */
    public Bundle updateBundle(Long id, Bundle bundleDetails) {
        Bundle bundle = bundleRepository.findById(id).orElse(null);
        if (bundle != null) {
            bundle.setTitle(bundleDetails.getTitle());
            bundle.setDescription(bundleDetails.getDescription());
            bundle.setQuantity(bundleDetails.getQuantity());
            bundle.setTicket(bundleDetails.getTicket());
            bundle.setDiscount(bundleDetails.getDiscount());
            return bundleRepository.save(bundle);
        }
        return null;
    }

    @Transactional
    /**
     * Delete a specific bundle from the database by its ID.
     *
     * @param id The ID of the bundle to delete.
     */
    public void deleteBundle(Long id) {
        bundleRepository.deleteById(id);
    }
}
