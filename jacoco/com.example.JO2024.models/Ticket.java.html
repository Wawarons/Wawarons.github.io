package com.example.JO2024.models;

import java.math.BigDecimal;
import java.time.Instant;
import java.util.ArrayList;
import java.util.List;

import jakarta.persistence.CascadeType;
import jakarta.persistence.Entity;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.GenerationType;
import jakarta.persistence.Id;
import jakarta.persistence.OneToMany;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.PositiveOrZero;
import jakarta.validation.constraints.Size;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Represents a ticket entity with details such as id, title, description,
 * price, quantity, availability status,
 * start date, and end date.
 * This class is annotated with Lombok annotations for generating constructors,
 * getters, setters, and builder methods.
 * It is also annotated with JPA annotations for entity mapping.
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Builder
@Setter
@Entity
public class Ticket {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private long id;

    @NotNull
    @Size(min = 3, max = 25)
    private String title;

    @Size(min = 5, max = 250)
    private String description;

    @NotNull
    @PositiveOrZero
    private BigDecimal price;

    @NotNull
    @PositiveOrZero
    private int quantity;

    @NotNull
    private boolean isAvailable;

    @NotNull
    private Instant startDate;

    @NotNull
    private Instant endDate;
    
    public String toString() {
        return "Ticket : " + this.id;
    }

    @Builder.Default
    @OneToMany(mappedBy = "ticket", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<Bundle> bundles = new ArrayList<Bundle>();

}
