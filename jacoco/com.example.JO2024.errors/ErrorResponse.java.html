package com.example.JO2024.errors;

import java.util.ArrayList;
import java.util.List;

import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Represents an error response.
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Setter
public class ErrorResponse {


    private String error;
    private List<String> details = new ArrayList<>();
    private int code;

    public ErrorResponse(String error, int code) {
        this.error = error;
        this.code = code;
    }

    public ErrorResponse(List<String> details, int code) {
        this.details = details;
        this.code = code;
    }

    public void setError(String error) {
        if (error != null && !error.isEmpty()) {
            this.error = error;
        } else {
            throw new IllegalArgumentException("Error cannot be null or empty");
        }
    }

    public String getError() {
        return error;
    }

    public void setDetails(List<String> details) {
        this.details = new ArrayList<>(details);
    }

    public List<String> getDetails() {
        return details;
    }

    public void setCode(int code) {
        if(code > 0) {
            this.code = code;
        }else {
            throw new IllegalArgumentException("Error cannot be a negative value.");
        }
    }

    public int getCode() {
        return code;
    }

}
